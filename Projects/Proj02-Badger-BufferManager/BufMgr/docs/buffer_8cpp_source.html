69 void BufMgr::allocBuf(FrameId & frame)
70 {
71 
72  //the worst case, we need to run two cycles to ensure all frames are unavalible
73  for(std::uint64_t i = 0; i < (2*numBufs); i++){
74 
75  //1.Adavance Clock Pointer
76  advanceClock();
77 
78  //2.load current frame status
79  BufDesc* curr = &bufDescTable[clockHand];
80 
81  //3.check if valid bit is set
82  if(curr->valid == false) {
83  //return the available frame, readPage
84  //and allocPage will take over hash and set
85  frame = clockHand;
86  return;
87  }
88 
89  //4.check if the refbit is set
90  if(curr->refbit == true){
91  //since refbit = true, clean it and continue the loop for the next frame
92  curr->refbit = false;
93  continue;
94  }
95  else{
96  //5.check if the page is pinned
97  if(curr->pinCnt != 0)
98  {
99  //page is pinned, continue the loop for the next frame
100  continue;
101  }
102  else{
103  //6.find an avaliable frame
104  //check if the page is dirty
105  if(curr->dirty == true){
106  //if dirty, flush the page back to the disk
107  curr->file->writePage(bufPool[clockHand]);
108  bufStats.accesses++;
109  bufStats.diskwrites++;
110  }
111  //clear hashtable, bufdesc table, and return the frameId
112  hashTable->remove(curr->file, curr->pageNo);
113  curr->Clear();
114  frame = clockHand;
115  return;
116  }
117 
118  }
119 
120  }
121 
122  //no avaliable frame is found in the loop, throw exception;
123  throw BufferExceededException();

192 void BufMgr::flushFile(const File* file)
193 {
194  // Scan through the buffer table
195  for (std::uint32_t i = 0; i < numBufs; i++) {
196  if (bufDescTable[i].file == file) {
197  // Throw BadBufferException if the page is invalid
198  if (!bufDescTable[i].valid)
199  throw BadBufferException(
200  bufDescTable[i].frameNo, bufDescTable[i].dirty,
201  bufDescTable[i].valid, bufDescTable[i].refbit);
202  // Throw PagePinnedException if the page is pinned
203  if (bufDescTable[i].pinCnt > 0)
204  throw PagePinnedException(file->filename(),
205  bufDescTable[i].pageNo,
206  bufDescTable[i].frameNo);
207  // Write to file if this page is dirty
208  if (bufDescTable[i].dirty) {
209  bufDescTable[i].file->writePage(bufPool[i]);
210  bufStats.accesses++;
211  bufStats.diskwrites++;
212  bufDescTable[i].dirty = false;
213  }
214  // Remove the page from hash table
215  hashTable->remove(file, bufDescTable[i].pageNo);
216  // Invoke the Clear() method of BufDesc for the page frame
217  bufDescTable[i].Clear();
218  }
219  }
